###  Максимальний набір тестів для відкритої євопейської понадпорогової процедури

-s openProcedure

###  Параметри, що впливають на вхідні дані тесту

-v mode:openeu               # Відкрита європеська понадпорогова процедура

-v number_of_items:2         # Кількість предметів закупівлі ( у лоті або у тендері, якщо безлотовий). Натуральне число.

-v number_of_lots:2          # Кількість лотів у тендері. Натуральне число.

-v tender_meat:True          # Неціновий показник на тендер. True or False

-v lot_meat:True             # Неціновий показник на перший лот. True or False

-v item_meat:True            # Неціновий показник на перший предмет. True or False

### Тести універсальні для усіх відкритих процедур

-i create_tender                 # Створення тендера  (обов'язково)
-i search_tender                 # Пошук тендера  (обов'язково)
-i tender_view                   # Відображення тендера  (обов'язково)
-i item_view                     # Відображення номеклатур із вхідних даних (обов'язково, якщо number_of_items > 0)
-i meat_view                     # Відображення нецінових показників із вхідних даних (обов'язково, якщо хоча б один із *_meat:True)
-i lot_view                      # Відображення лотів із вхідних даних (обов'язково, якщо number_of_lots > 0)

-i add_tender_doc                # Добавлення документації до тендера
-i add_lot_doc                   # Добавлення документації для першого лоту
-i lot_modify                    # Редагування першого лот
-i add_lot                       # Створення нового лоту
  -i delete_lot                  # Видалення останнього лоту
-i add_item                      # Створення нової номеклатури
  -i delete_item                 ## Видалення нової номеклатури
-i add_tender_meat               # Добавлення нецінового показника на тендер
  -i delete_tender_meat          ## Видалення нецінового показника на тендер
-i add_lot_meat                  # Добавлення нецінового показника на перший лот
  -i delete_lot_meat             ## Видалення нецінового показника на перший лот
-i add_item_meat                 # Добавлення нецінового показника на перший предмет
  -i delete_item_meat            ## Видалення нецінового показника на перший предмет

-i question_to_tender            # Запитання тендер
  -i answer_question_to_tender   ## Відповідь на запитання на тендер
-i question_to_item              # Запитання на предмет
  -i answer_question_to_item     ## Відповідь на запитання на предмет
-i question_to_lot               # Запиння на лот
  -i answer_question_to_lot      ## Відповідь на запитання на лот

# -i bid_before_bid_period       # Неможливість подати пропозицію по початки періоду подачі пропозиції
-i bid_without_related_lot       # Подання пропозицій без прив'язки до лоту
-i bid_without_parameters        # Неможливість подати пропозицію без нецінових показників
-i provider_bid                  # Подання пропозиції першим учасником
  -i provider_bid_modify         ## Редагування пропозиції першим учасником
  -i add_bid_doc    ## Додадання документації для пропозиції першим учасником
# -i provider_bid_canceled       ## Видалення пропозиції першим учасником
-i provider1_bid                 # Подання пропозиції другим учасником


### Тести специфічні для відкритих понадпорогових процедур

-i open_tender_view                       # Відображення полів тендера специфічних для понадпороговї процедури
-i open_modify_tender_in_tendering_perion # Редагування тендера у період подачі пропозиції
  -i open_confirm_first_bid               # Підтвердити пропозицію першим учасником після редагування тендера
  -i open_confirm_second_bid              # Підтвердити пропозицію другим учасником після редагування тендера

### Тести специфічні для відкритої європейської понадпорогової процедури

-i openeu_provider_bid_private_doc        # Перетворення публічного документу пропозиції на приватний першим учасником
-i openeu_provider_bid_financial_doc      # Завантаження фінансового документу у пропозиції першим учасником
-i openeu_provider_bid_qualification_doc  # Завантаження кваліфікаційного документу у пропозицію першим учасником
-i openeu_provider_bid_eligibility_doc    # Завантаження документу для критерії прийнятності у пропозицію першим учасником

## Прекваліфікація відкритої європейської понадпорогової процедури

-i pre-qualification_add_doc_to_tender    # Неможливість добавити документацію до тендера під час прекваліфікації
-i pre-qualification_add_doc_to_lot       # Неможливість добавити документацію до лоту під час прекваліфікації

-i pre-qualification_view                 # Відображення основних даних прекваліфікації

-i pre-qualification_fist_bid_doc         # Завантаження документу кваліфікації до пропозиції першого учасника
-i pre-qualification_approve_first_bid    # Підтвердження кваліфікації пропозиції першого учасника

-i pre-qualification_second_bid_doc                    # Завантаження документу кваліфікації до пропозиції другого учасника
-i pre-qualification_reject_second_bid                 # Відхилення кваліфікації пропозиції другого учасника
  -i pre-qualification_cancel_second_bid_qualification # Скасування рішення кваліфікації пропозиції другого учасника
-i pre-qualification_approve_second_bid                # Підтвердження кваліфікації пропозиції другого учасника

-i pre-qualification_approve_qualifications            # Підтвердження кваліфікації усіх пропозицій
